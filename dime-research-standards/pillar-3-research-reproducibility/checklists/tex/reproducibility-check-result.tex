		\section*{Part 1: Computational Reproducibility}

		\subsection{Required}

		\noindent
		\begin{Form}
			\dimeCheckBox{ All code runs completely on a new computer.}
		\end{Form}
	
		\noindent
		\begin{Form}
			\dimeCheckBox{ The master script re-creates all results as they appear in the paper.}
		\end{Form}


		\subsection{Recommended}
		\begin{Form}
			\dimeCheckBox{ All in-text numerical citations that are not drawn directly from tables and figures are reproduced by code provided in the replication package.}
		\end{Form}

		\noindent
		\begin{Form}
			\dimeCheckBox{ Scripts for data cleaning, variable creation, and analysis are separate.}
		\end{Form}

		\noindent
		\begin{Form}
			\dimeCheckBox{ Analysis scripts do not include any data cleaning or variable creation, unless necessary for the creation of a table or graphic.}
		\end{Form}

		\noindent
		\begin{Form}
			\dimeCheckBox{ All code is organized, commented, and formatted such that a new reader or reviewer can easily identify the code and raw outputs that produce each final output or manuscript exhibit.}
		\end{Form}
	
		\begin{adjustwidth}{0.6cm}{}
		\textbf{Important:} Good organization will allow DIME Analytics to provide better feedback on individual exhibits (whether or not there are reproducibility issues), and will allow technical reviewers to evaluate whether code correctly implements statistical and econometric processes.
		\end{adjustwidth}

		\noindent
		\begin{Form}
			\dimeCheckBox{ All intermediate data sets can be reproduced from the raw data.}
		\end{Form}

		\noindent
		\begin{Form}
			\dimeCheckBox{ All final data sets can be reproduced from the data submitted.}
		\end{Form}
