		\section*{Data cleaning tasks}
  This checklist lists important factors to consider while reviewing your code review partner's \textbf{cleaning} scripts/code. Please fill this checklist, and submit it as an attachment when you submit \href{https://survey.wb.surveycto.com/collect/code_review_summary?caseid=}{this detailed form.}
		
		\begin{itemize}
			\setlength\itemsep{-0.1em}
			\item[] \dimeCheckBox{Cleaning scripts do not create indicators (this should be done in a separate construction script)}
              \item[] \dimeCheckBox{Are any changes being made to observations values in the cleaning do-files?}
              \item[]\dimeCheckBox{\textbf{Check duplicates:}} 
            
            \begin{itemize}
                \item[] \dimeCheckBox{The code checks for duplicates in ID variables and other important variables?}
                \item[] \dimeCheckBox{If so, are the duplicates fixed?}
                \item[] \dimeCheckBox{In cases where duplicates are not resolved, is there a comment explaining next steps?}
            \end{itemize}
			\item[] \dimeCheckBox{\textbf{Check merges:}} 
            
            \begin{itemize}
                \item[] \dimeCheckBox{Are any observations dropped?}
                \item[] \dimeCheckBox{If so, is there a clear justification for that?}
                \item[] \dimeCheckBox{If any observations didn't match, is that explained in the comments?}
            \end{itemize}
            \item[] \dimeCheckBox{\textbf{Check labels:}} 
            
            \begin{itemize}
                \item[] \dimeCheckBox{Are all variables correctly labelled?}
                \item[] \dimeCheckBox{If so, are the labels consistent?}
            \end{itemize}

            \dimeCheckBox{\textbf{Check missing values:}} 
            \begin{itemize}
			\item[] \dimeCheckBox{Are missing values coded consistently?}
                \item[] \dimeCheckBox{Are extended missing values used? (e.g. \texttt{.d} for \textbf{Do not know}, \texttt{.r} for \textbf{Refuse to answer}, etc.}
            \end{itemize}
                       
			
		\end{itemize}
