		\section*{Analysis tasks}
  This checklist lists important factors to consider while reviewing your code review partner's \textbf{analysis} scripts/code. Please fill this checklist, and submit it as an attachment when you submit \href{https://survey.wb.surveycto.com/collect/code_review_summary?caseid=}{this detailed form.}

		
		\begin{itemize}
			\setlength\itemsep{-0.1em}
			\item[] \dimeCheckBox{No variables are created in analysis scripts unless very specific to a particular output}
           

                \item[] \dimeCheckBox{Code is modular, such that individual outputs can be run in any order}
                
                \item[] \dimeCheckBox{Research decisions are well documented in the code (e.g. sample used, treatment of missing variables)}
                
                
                \item[] \dimeCheckBox{The functions used are appropriate for the analysis being performed}
                
                \item[] \dimeCheckBox{Categorical variables are used correctly (i.e. labeled integers are not used as continuous variables)}


                \item[] \dimeCheckBox{Does the code produce any outputs?}

                \begin{itemize}
                \item[] \dimeCheckBox{Outputs are exported in a reproducible manner, and manual formatting is very limited}
                \end{itemize}

            \end{itemize}
